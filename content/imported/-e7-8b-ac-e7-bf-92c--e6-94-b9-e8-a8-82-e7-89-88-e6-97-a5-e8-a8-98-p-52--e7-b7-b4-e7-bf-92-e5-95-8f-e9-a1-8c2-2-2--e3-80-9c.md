    #include 

    using namespace std;

    class box {
      double height;
      double width;
      double breadth;

    public:
      box(double _height,
          double _width,
          double _breadth);
      void vol();
    };

    box::box(double _height, double _width, double _breadth)
    {
      height = _height;
      width = _width;
      breadth = _breadth;
    }

    void box::vol()
    {
      printf("vol:%f\n", height * width * breadth);
    }

    int main()
    {
      box b(10.0, 20.0, 30.0);

      b.vol();

      return 0;
    }
